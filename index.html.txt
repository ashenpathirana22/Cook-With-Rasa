        function renderRecipes(filter = 'all') {
            const grid = document.getElementById('recipeGrid');
            grid.innerHTML = '';
            
            const filtered = filter === 'all' ? recipes : recipes.filter(r => r.category === filter);
            
            filtered.forEach((recipe, index) => {
                const card = document.createElement('div');
                card.className = 'recipe-card';
                card.style.animation = `slideUp 0.6s ease ${index * 0.1}s forwards`;
                card.style.opacity = '0';
                card.onclick = () => openModal(recipe);
                
                card.innerHTML = `
                    <div class="recipe-image" data-emoji="${recipe.emoji}">
                        <span class="recipe-badge">${recipe.category}</span>
                    </div>
                    <div class="recipe-content">
                        <h3 class="recipe-title">${recipe.title}</h3>
                        <div class="recipe-meta">
                            <span>‚è± ${recipe.time}</span>
                            <span>üìä ${recipe.difficulty}</span>

                        <div class="recipe-footer">
                            <span class="difficulty">${recipe.difficulty}</span>
                            <span class="rating">‚òÖ ${recipe.rating}</span>
                        </div>
                    </div>
                `;
                
                grid.appendChild(card);
            });
        }
        
        function filterRecipes(category) {
            document.querySelectorAll('.filter-btn').forEach(btn => {
                btn.classList.remove('active');
                if(btn.textContent.toLowerCase().includes(category) || (category === 'all' && btn.textContent === 'All')) {
                    btn.classList.add('active');
                }
            });
            renderRecipes(category);
        }
        
        function openModal(recipe) {
            const modal = document.getElementById('recipeModal');
            const body = document.getElementById('modalBody');
            
            body.innerHTML = `

                    <h2>${recipe.title}</h2>
                    <div style="color: #666; margin-top: 0.5rem;">

            document.body.style.overflow = 'hidden';

        function closeModal() {
            document.getElementById('recipeModal').style.display = 'none';
            document.body.style.overflow = 'auto';
        }
